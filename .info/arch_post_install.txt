############

alias dotfiles='/usr/bin/git --git-dir=$HOME/.dotfiles/ --work-tree=$HOME'

rm $HOME/.bash_profile
rm $HOME/.bashrc
dotfiles checkout

dotfiles config --local status.showUntrackedFiles no
